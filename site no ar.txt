Código de Conexão em PythonO código no seu arquivo do Streamlit permanece limpo e seguro, usando a biblioteca padrão psycopg2 (certifique-se de incluir psycopg2-binary no seu requirements.txt):pythonimport streamlit as st
import psycopg2

def conectar_supabase():
    conn = psycopg2.connect(
        host=st.secrets["postgres"]["host"],
        database=st.secrets["postgres"]["database"],
        user=st.secrets["postgres"]["user"],
        password=st.secrets["postgres"]["password"],
        port=st.secrets["postgres"]["port"]
    )
    return conn

# Testando a conexão
try:
    conn = conectar_supabase()
    st.success("Conectado ao Supabase com sucesso!")
except Exception as e:
    st.error(f"Erro ao conectar: {e}")
